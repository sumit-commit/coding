int oneIteration(vector<int> A)
{
    int max = INT_MIN, smax = -1;

    for(int i=0;i<A.size();i++)
    {
        if(A[i] > max)
        {
            smax = max;
            max = A[i];
        }
        else if(A[i] > smax)
        {
            smax = A[i];
        }
    }

    return max+smax;
}

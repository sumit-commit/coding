string sum(int size, int arr[], int target)
{
    if(size == 0)
    {
        return "Not found";
    }
    
    if(arr[0] == target)
    {
        return "Found ";
    }
    
    string ans = sum(size-1, arr+1, target);
    return ans;
}
int main()
{
    int arr[5] = {3,2,5,1,6};
    int target;
    
    cout<<"Enter item to search = ";
    cin>>target;
    
    cout<<sum(5, arr, target);
    
    return 0;
}

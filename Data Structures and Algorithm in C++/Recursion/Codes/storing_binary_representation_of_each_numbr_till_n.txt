string converting_to_bin(int n) {

    if(n == 1)
    {
       string str = "1";
       return str;
    }

    string str = converting_to_bin(n/2);

    str += to_string(n%2);
    
    return str;

}

vector<string> generateBinaryNumbers(int n)
{
    if(n == 0)
    {
        vector<string> answer;
        return answer;
    }
    
    vector<string> answer = generateBinaryNumbers(n-1);
   
    answer.push_back(converting_to_bin(n));

    return answer;
    
}
